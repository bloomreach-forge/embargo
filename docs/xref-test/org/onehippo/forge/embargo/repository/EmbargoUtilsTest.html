52      @Test
53      public void testEmbargoGetExpirationDateAsNotNull() throws Exception {
54          final Calendar instance = Calendar.getInstance();
55  
56          Node mockHandleNode = createMock(Node.class);
57          Node requestNode = createMock(Node.class);
58          Node triggersNode = createMock(Node.class);
59          MockProperty triggerFireTimeProperty = new MockProperty(HIPPOSCHED_TRIGGERS_DEFAULT_PROPERTY_FIRETIME);
60          triggerFireTimeProperty.setValue(instance);
61  
62          expect(mockHandleNode.isNodeType(HippoNodeType.NT_HANDLE)).andReturn(true);
63          expect(mockHandleNode.hasNode(EMBARGO_SCHEDULE_REQUEST_NODE_NAME)).andReturn(true);
64          expect(mockHandleNode.getNode(EMBARGO_SCHEDULE_REQUEST_NODE_NAME)).andReturn(requestNode);
65          expect(requestNode.hasNode(HIPPOSCHED_TRIGGERS_DEFAULT)).andReturn(true);
66          expect(requestNode.getNode(HIPPOSCHED_TRIGGERS_DEFAULT)).andReturn(triggersNode);
67  
68          expect(triggersNode.hasProperty(HIPPOSCHED_TRIGGERS_DEFAULT_PROPERTY_FIRETIME)).andReturn(true);
69          expect(triggersNode.getProperty(HIPPOSCHED_TRIGGERS_DEFAULT_PROPERTY_FIRETIME)).andReturn(triggerFireTimeProperty);
70  
71          replay(mockHandleNode, requestNode, triggersNode);
72  
73          final Calendar embargoExpirationDate = EmbargoUtils.getEmbargoExpirationDate(mockHandleNode);
74          assertEquals(instance, embargoExpirationDate);
75      }
